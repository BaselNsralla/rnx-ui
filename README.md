A UI Lib for React Native

## Start

### 1. Install

```js
npm install 'rnx-ui' --save
```

### 2. Use

```js
import React, {
  Component,
} from 'react';
import SlideMenu from 'rnx-ui/SlideMenu';

function Demo(props) {
  return <SlideMenu />;
}
```
